\def\callunitipa#1{%
	\immediate\write18{unitipa.sh "#1" > tmp_unicode-tipa.tex}%
	\immediate\openin\tmpfile=tmp_unicode-tipa.tex%
	\read\tmpfile to \tmpcontent%
	\immediate\closein\tmpfile%
	\tmpcontent%
}
